2. 占比加修改

残基与表面的最小距离gmx mindist
分析最后100ns的占比 ($1 >= 200) 可以修改
按照距离大小给予权重 0.35 0.8 1.6 .. (nm)

awk '!/^(#|@)/ && $1 >= 200 {
  for (j = 2; j <= NF; j++) {
    if ($j < 0.2) {
      freq[j] += 1;
    } else if ($j < 0.35) {
      freq[j] += 0.9;
    } else if ($j < 0.5) {
      freq[j] += 0.7;
    } else if ($j < 0.9) {
      freq[j] += 0.5;
    } else if ($j < 1.6) {
      freq[j] += 0.3;
    } else {
      freq[j] += 0;
    }
  }
  count++;
} END {
  for (i in freq) {
    printf("%5.2f\n", freq[i] / count);
  }
}' mindist.xvg > occ.dat
